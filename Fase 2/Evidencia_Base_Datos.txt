create table
  public.dueño (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone not null default now(),
    nombre character varying null,
    fecha_nacimiento date null,
    constraint dueño_pkey primary key (id)
  ) tablespace pg_default;

create table
  public.mascota (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone not null default now(),
    nombre character varying null,
    raza character varying null,
    edad smallint null,
    color1 character varying null,
    color2 character varying null,
    color3 character varying null,
    especie character varying null,
    tamano character varying null,
    estado smallint null,
    descripcion text null,
    id_dueno bigint null,
    foto_perfil text null,
    constraint mascota_pkey primary key (id)
  ) tablespace pg_default;


create table
  public.mascota_encontrada (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone not null default now(),
    raza character varying null,
    color1 character varying null,
    color2 character varying null,
    color3 character varying null,
    especie character varying null,
    tamano character varying null,
    estado smallint null,
    ubicacion character varying null,
    descripcion text null,
    constraint mascota_perdida_pkey primary key (id)
  ) tablespace pg_default;
